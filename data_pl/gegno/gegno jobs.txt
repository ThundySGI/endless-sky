mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "Zbierz 10 ton aluminium"
	description "Wydobadz i zbierz 10 ton aluminium, a nastepnie przywiez mineraly do Gegno w punkcie <origin>."
	source "Tschyss"
	to offer
		has "Gegno Asteroid Mining Prologue: done"
		random < 10
	to accept
		"cargo space" > 10
	on visit
		dialog "Wrociles do <origin>, ale nie masz 10 ton aluminium w ladowni. Upewnij sie, ze wydobyles 10 ton aluminium i zapakowales je do ladowni przed ladowaniem."
	on complete
		"gegno: mining jobs" ++
		payment 36000
		outfit "Aluminum" -10
		dialog phrase "gegno mining payment dialog"
