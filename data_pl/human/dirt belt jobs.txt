mission "Drought Relief"
	name `Pomoc w zwiazku z susza do <planet>`
	job
	repeat
	description `Swiat rolnictwa <destination> doswiadcza obecnie suszy. Dostarcz <cargo> na planete do <date> za <payment>.`
	cargo "drought relief supplies" 25 2 .05
	deadline
	to offer
		random < 10
	source
		government "Republic"
		not attributes "farming"
	destination
		distance 3 10
		attributes "dirt belt"
		attributes "farming"
	on visit
		dialog phrase "generic cargo on visit"
	on complete
		payment 10000 160
		dialog "Zrzucasz ladunek <commodity> i odbierasz zaplate w wysokosci <payment>. Grupa <planet> obywateli dziekuje ci i zegna cie."


mission "To Remembrance Day celebration [0]"
	name `Obchody Dnia Pamieci`
	job
	repeat
	description `Przynies <fare> do <destination> do <date> na obchody Dnia Pamieci 19 lutego. Zaplata wynosi <payment>.`
	passengers 4 4 .5
	deadline
	to offer
		or
			and
				random < 20
				month == 12
				day >= 21
			and
				random < 30
				month == 1
			and
				random < 10
				month == 2
				day < 19
	to fail
		month * 100 + day >= 219
	source
		attributes "dirt belt"
	destination
		distance 3 15
		attributes "dirt belt"
	on visit
		dialog phrase "generic passenger on visit"
	on complete
		payment 1000 180
		dialog "Zyczysz swojemu <passengers> wszystkiego najlepszego na <planet>. Dzieci uciekaja z twojego statku pomalowane na zielono, a rodzice placa ci <payment>."
