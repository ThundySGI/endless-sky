mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "<NM-625>"
	description "<DS-2445>"
	source "Tschyss"
	to offer
		has "Gegno Asteroid Mining Prologue: done"
		random < 10
	to accept
		"cargo space" > 10
	on visit
		dialog "<DL-524>"
	on complete
		"gegno: mining jobs" ++
		payment 36000
		outfit "Aluminum" -10
		dialog phrase "gegno mining payment dialog"
